## 77. Combinations


Given two integers n and k, return all possible combinations of k numbers out of 1 ... n.

You may return the answer in any order.
 

Example 1:

Input: n = 4, k = 2
Output:
[
  [2,4],
  [3,4],
  [2,3],
  [1,2],
  [1,3],
  [1,4],
]

Example 2:

Input: n = 1, k = 1
Output: [[1]]
 

Constraints:

* 1 <= n <= 20
* 1 <= k <= n

```c++
class Solution {
 public:
  vector<vector<int>> combine(int n, int k) {
      std::vector<std::vector<int>> res;
      std::vector<int> temp;
      std::function<void(int)> create = [&](int begin){
          if(temp.size() + n - begin + 1 < k) {
              return;
          }
          if(temp.size() == k){
              res.push_back(temp);
              return;
          }
          temp.push_back(begin);
          create(begin+1);
          temp.pop_back();
          create(begin+1);
      };
      create(1);
      return res;
  }
};
```

class Solution {
public:
    vector<vector<int>> combine(int n, int k) {
        vector<vector<int>> a;
        vector<int> b(k,0);
        int i=0;
        while (i >= 0)
        {
            b[i]++;
            if (b[i] > n) i--;
            else if (i == k - 1) a.push_back(b);
            else
            {
                i++;
                b[i] = b[i - 1];
            }
        }
        return a;
    }
};

class Solution {
public:
    int Hamming_weight(int number)
    {
        int count = 0;
        while (number != 0)
        {
            number &= (number - 1);
            count++;
        }
        return count;
    }
public:
    vector<vector<int>> combine(int n, int k) {
        vector<vector<int>> ans;
        for (int i = 0; i < (1 << n); i++) {
            if (Hamming_weight(i) == k) {
                vector<int> v;
                int k = i;
                for (int j = 0; j < n; j++) {
                    if ((k & 1) == 1) {
                        v.push_back(j + 1);
                    }
                    k = (k >> 1);
                }
                ans.push_back(v);
            }
        }
        return ans;
    }
};
