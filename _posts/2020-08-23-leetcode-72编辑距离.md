## 72.Edit Distance

Given two words word1 and word2, find the minimum number of operations required to convert word1 to word2.

You have the following 3 operations permitted on a word:

Insert a character

Delete a character

Replace a character

Example 1:

Input: word1 = "horse", word2 = "ros"

Output: 3

Explanation: 
horse -> rorse (replace 'h' with 'r')
rorse -> rose (remove 'r')
rose -> ros (remove 'e')

Example 2:

Input: word1 = "intention", word2 = "execution"

Output: 5

Explanation: 
intention -> inention (remove 't')

inention -> enention (replace 'i' with 'e')

enention -> exention (replace 'n' with 'x')

exention -> exection (replace 'n' with 'c')

exection -> execution (insert 'u')

```c++
class Solution {
public:
    int minDistance(string word1, string word2) {
        int n = word1.length();
        int m = word2.length();

        // 有一个字符串为空串
        if (n * m == 0) return n + m;

        // DP 数组
        int D[n + 1][m + 1];

        // 边界状态初始化
        for (int i = 0; i < n + 1; i++) {
            D[i][0] = i;
        }
        for (int j = 0; j < m + 1; j++) {
            D[0][j] = j;
        }

        // 计算所有 DP 值
        for (int i = 1; i < n + 1; i++) {
            for (int j = 1; j < m + 1; j++) {
                int left = D[i - 1][j] + 1;
                int down = D[i][j - 1] + 1;
                int left_down = D[i - 1][j - 1];
                if (word1[i - 1] != word2[j - 1]) left_down += 1;
                D[i][j] = min(left, min(down, left_down));

            }
        }
        return D[n][m];
    }
};
```
递归解法
```c++
  int find(string& word1, string& word2, int i, int j) {
    if (i == -1) {
      return j + 1;
    }
    if (j == -1) {
      return i + 1;
    }
    if (word1.at(i) == word2.at(j)) {
      return find(word1, word2, i - 1, j - 1);
    } else {
      int temp = find(word1, word2, i, j - 1) + 1;       ///< 插入
      int temp1 = find(word1, word2, i - 1, j) + 1;      ///< 删除
      int temp2 = find(word1, word2, i - 1, j - 1) + 1;  ///< 替换
      return std::min(temp, std::min(temp1, temp2));
    }
  }
  int minDistance(string word1, string word2) {
    return find(word1, word2, word1.size() - 1, word2.size() - 1);
  }
```

```c++
  int minDistance(string word1, string word2) {
    int len1 = word1.size();
    int len2 = word2.size();
    std::vector<int> temp(len2 + 1);
    std::vector<std::vector<int>> vec(len1 + 1, temp);
    for (int i = 1; i <= len1; i++) {
      vec.at(i).at(0) = i;
    }
    for (int j = 1; j <= len2; j++) {
      vec.at(0).at(j) = j;
    }

    for (int i = 1; i <= len1; i++) {
      for (int j = 1; j <= len2; j++) {
        if (word1.at(i - 1) == word2.at(j - 1)) {
          vec.at(i).at(j) = vec.at(i - 1).at(j - 1);
        } else {
          auto temp = vec.at(i - 1).at(j) + 1;       ///< 插入
          auto temp1 = vec.at(i).at(j - 1) + 1;      ///< 删除
          auto temp2 = vec.at(i - 1).at(j - 1) + 1;  ///< 替换
          vec.at(i).at(j) = std::min(temp, std::min(temp1, temp2));
        }
      }
    }
    return vec.at(len1).at(len2);
  }
};
```
