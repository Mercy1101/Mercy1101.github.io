## 206. 反转链表

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(head == nullptr) return nullptr;
        auto cur = head;
        ListNode* before = nullptr;
        ListNode* next = head->next;
        while(cur != nullptr) {
            next = cur->next;
            cur->next = before;  
            before = cur;  
            if(!next)  {
                break;
            } 
            cur = next;
        }
        return cur;
    }
};
```
